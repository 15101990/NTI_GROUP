


#include "LED.h"
#include "BUZZER.h"
#include "P_B.h"
#include "SSD.h"
#include "LCD.h"
#include "KEY_PAD.h"
#include <avr/interrupt.h>
#define F_CPU 16000000UL
#include <util/delay.h>



int main(void)
{
	LCD_INIT();
	KEY_PAD_INIT();
	while(1)
	{
		u8 x = KEY_PAD_READ();
		if(x != 0)
		{
			LCD_CLR();
			LCD_WRITE_CHR(x);
			x = 0;
		}
	}
}
